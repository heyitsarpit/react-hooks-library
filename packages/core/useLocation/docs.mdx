# useLocation

Reactive browser location.

## Usage

```tsx
import { useLocation } from '@react-hooks-library/core'

export function Demo() {
  const router = useLocation()

  return (
    <div>
      <pre>
        <code>{JSON.stringify(router, null, 2)}</code>
      </pre>
    </div>
  )
}
```
